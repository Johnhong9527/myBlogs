# 点击事件

该点击事件,可作用任意元素中

// 图片点击事件
Widget image_on_tap(info) {
  // print(info[0]);
  // print(93);
  return GestureDetector(
    onTap: () {
    },
    child: Image.asset(
      '${info["file"]}.png',
      width: 100.0,
      height: 100.0,
    ),
  );
}
